class Solution {
    public void rotate(int[] nums, int k) {
          int n = nums.length;
            k = k % n;
            int[] tmp = new int[n];
            System.arraycopy(nums, 0, tmp, 0, n);
            System.arraycopy(tmp, n - k, nums, 0, k);
            System.arraycopy(tmp, 0, nums, k, n - k);
    }
}
